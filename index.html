enrollment_comparison.sql
-- Compare enrollment year over year
SELECT
  YEAR(reg.add_date) AS enrollment_year,
  COUNT(DISTINCT reg.id_num) AS students,
  SUM(reg.credit_hrs) AS total_credits
FROM student_crs_hist reg
WHERE reg.stud_div = 'UG'
  AND reg.transaction_sts = 'C'
GROUP BY YEAR(reg.add_date)
ORDER BY enrollment_year DESC;
